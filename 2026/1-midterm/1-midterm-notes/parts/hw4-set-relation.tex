\section{作业 4：集合与关系}

\begin{problem}[关系的运算]
  请证明,
  \[
    (X \cap Y) \circ Z \subseteq (X \circ Z) \cap (Y \circ Z).
  \]
  请举例说明, $\subseteq$ 不能换成 $=$。
\end{problem}

\begin{proof}
  对任意 $(a, c)$,
  \begin{align*}
    (a, c) \in (X \cap Y) \circ Z
      &\iff \exists b\; (a, b) \in Z \land (b, c) \in X \cap Y \\
      &\iff \exists b\; \big((a, b) \in Z \land (b, c) \in X \land (b, c) \in Y\big) \\
      &\;\red{\implies} \exists b\; \big((a, b) \in Z \land (b, c) \in X\big)
        \land \exists b\; \big((a, b) \in Z \land (b, c) \in Y\big) \\
      &\iff (a, c) \in X \circ Z \land (a, c) \in Y \circ Z \\
      &\iff (a, c) \in (X \circ Z) \cap (Y \circ Z).
  \end{align*}
  故
  \[
    (X \cap Y) \circ Z \subseteq (X \circ Z) \cap (Y \circ Z).
  \]

  下面给出严格包含的反例。取
  \[
    X = \set{(2, 4)},
    \qquad
    Y = \set{(3, 4)},
    \qquad
    Z = \set{(1, 2), (1, 3)}.
  \]
  则
  \[
    X \cap Y = \emptyset,
    \qquad
    (X \cap Y) \circ Z = \emptyset.
  \]
  另一方面,
  \[
    X \circ Z = \set{(1, 4)},
    \qquad
    Y \circ Z = \set{(1, 4)},
  \]
  从而
  \[
    (X \circ Z) \cap (Y \circ Z) = \set{(1, 4)}.
  \]
  因此
  \[
    (X \cap Y) \circ Z = \emptyset \subsetneq \set{(1, 4)} = (X \circ Z) \cap (Y \circ Z).
  \]
\end{proof}

\begin{problem}[关系的性质]
  请证明,
  \[
    R \text{ 是对称且传递的 } \iff R = R^{-1} \circ R
  \]
\end{problem}

\begin{proof}
	先证
  \[
    R \text{ 是对称且传递的 } \implies R = R^{-1} \circ R.
  \]

  假设 $R$ 是对称且传递的。
  因为 $R$ 是对称的, 所以
  \[
    R = R^{-1}.
  \]
  因为 $R$ 是传递的, 所以
  \[
    R \circ R \subseteq R.
  \]
  故,
  \[
    R^{-1} \circ R = R \circ R \subseteq R.
  \]
  又, 任取 $(a, b) \in R$。
  因为 $R$ 是对称的, 所以 $(b, a) \in R$。
  又因为 $R$ 是传递的, 所以 $(b, b) \in R$ 且 $(b, b) \in R^{-1}$。
  因此, $(a, b) \in R^{-1} \circ R$。
	故, $R \subseteq R^{-1} \circ R$。

  再证
  \[
     R = R^{-1} \circ R \implies R \text{ 是对称且传递的 }.
  \]

  假设 $R = R^{-1} \circ R$。
  \[
    R^{-1} = (R^{-1} \circ R)^{-1} = R^{-1} \circ R = R,
  \]
  所以, $R$ 是对称的。

  \[
    R = R^{-1} \circ R = R \circ R,
  \]
  所以, $R$ 是传递的。
\end{proof}
