\begin{frame}[allowframebreaks]{作业 4：集合与关系}

\begin{problem}[关系的运算]
  请证明
  \[(X \cap Y) \circ Z \subseteq (X \circ Z) \cap (Y \circ Z),\]
  并举例说明不能换成等号。
\end{problem}

\pause
\begin{proof}
  取任意 $(a,c)$, 由定义可推导出包含关系, 并给出反例
  \(X=\{(2,4)\},\; Y=\{(3,4)\},\; Z=\{(1,2),(1,3)\}\).
\end{proof}

\pause
\begin{problem}[关系的性质]
  请证明 $R$ 对称且传递当且仅当 $R = R^{-1} \circ R$。
\end{problem}

\begin{proof}
  见讲义中的分步证明, 首先从对称+传递推出等式, 再反向证明等价性。
\end{proof}
