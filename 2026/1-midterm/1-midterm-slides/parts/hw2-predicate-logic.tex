\begin{frame}[allowframebreaks]{作业 2：谓词逻辑}

\begin{problem}[一阶谓词逻辑: 替换]
  请证明, 对于任意公式 $\alpha$ 与其中的任意变元 $x$,
  $x$ 都可以在 $\alpha$ 中无冲突地替换自己。
\end{problem}
\pause
\begin{solution}
  证明命题:
  \[P(\alpha): \text{对任意变元 } x,\; x \text{ 都可以在 } \alpha \text{ 中无冲突地替换 } x.\]

  对公式 $\alpha$ 作结构归纳。

  \textbf{基础情况.} 若 $\alpha$ 是原子公式, 则按 ``无冲突替换'' 的定义, 基本成立。

  \textbf{归纳步骤.} 请参见讲义中的详细归纳论证, 分为否定、复合、量词三种情形。
\end{solution}

\pause
\begin{problem}[一阶谓词逻辑: 前束范式]
  将以下公式化归成前束范式:
  \[\forall x\; \exists y\; P(x, y) \to \exists x\; Q(x).\]
\end{problem}

\begin{solution}
  先重命名, 再依规则 R5, R6, R4 进行迁移, 可得前束范式:
  \[\exists x\; \forall y\; \exists z\; (P(x, y) \to Q(z)).\]
\end{solution}
